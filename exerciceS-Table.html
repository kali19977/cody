EXERCICE 1 :

Écrire un programme en C qui demande à l'utilisateur de saisir 5 nombres entiers et les stocke dans un tableau. Ensuite, affichez ces nombres dans l'ordre inverse.
SOLUTION :
#include <stdio.h>
int main() {
int tableau[5];
int i;
printf("Veuillez saisir 5 nombres entiers :\n");
// Saisie des nombres et stockage dans le tableau
for (i = 0; i < 5; i++) {
printf("Nombre %d : ", i + 1);
scanf("%d", &tableau[i]);
}
printf("Les nombres dans l'ordre inverse sont :\n");
// Affichage des nombres dans l'ordre inverse
for (i = 4; i >= 0; i--) {
printf("%d ", tableau[i]);
}
return 0;
}
EXPLICATION :
Tout d'abord,
dans ce programme, nous utilisons un tableau de taille 5 pour stocker les nombres entiers saisis par l'utilisateur. Nous utilisons une boucle for pour demander à l'utilisateur de saisir les nombres et les stocker dans le tableau. Ensuite, nous utilisons une autre boucle for pour parcourir le tableau en ordre inverse et afficher les nombres.

EXERCICE 2 :

Écrire un programme en C qui lit un tableau de 10 entiers et calcule la somme de tous les éléments du tableau.
SOLUTION :
#include <stdio.h>
int main() {
int tableau[10];
int i;
int somme = 0;
printf("Veuillez saisir 10 nombres entiers :\n");
// Saisie des nombres et calcul de la somme
for (i = 0; i < 10; i++) {
printf("Nombre %d : ", i + 1);
scanf("%d", &tableau[i]);
somme += tableau[i];
}
printf("La somme des éléments du tableau est : %d\n", somme);
return 0;
}
EXPLICATION :
Tout d'abord,
dans ce programme, nous utilisons un tableau de taille 10 pour stocker les nombres entiers saisis par l'utilisateur. Nous utilisons une boucle for pour demander à l'utilisateur de saisir les nombres, les stocker dans le tableau et calculer la somme en ajoutant chaque élément au fur et à mesure. Enfin, nous affichons la somme calculée à l'utilisateur.

EXERCICE 3 :

Écrire un programme qui demande à l'utilisateur de saisir 10 nombres entiers et les stocke dans un tableau. Ensuite, trouvez et affichez le plus grand nombre dans le tableau.
SOLUTION :
#include <stdio.h>
int main() {
int tableau[10];
int i;
int plusGrand;
printf("Veuillez saisir 10 nombres entiers :\n");
// Saisie des nombres
for (i = 0; i < 10; i++) {
printf("Nombre %d : ", i + 1);
scanf("%d", &tableau[i]);
}
// Recherche du plus grand nombre
plusGrand = tableau[0];
for (i = 1; i < 10; i++) {
if (tableau[i] > plusGrand) {
plusGrand = tableau[i];
}
}
printf("Le plus grand nombre dans le tableau est : %d\n", plusGrand);
return 0;
}
EXPLICATION :
Tout d'abord,
Dans ce programme, nous utilisons un tableau de taille 10 pour stocker les nombres entiers saisis par l'utilisateur. Nous utilisons une boucle for pour demander à l'utilisateur de saisir les nombres et les stocker dans le tableau. Ensuite, nous utilisons une autre boucle for pour trouver le plus grand nombre dans le tableau en comparant chaque élément avec le plus grand nombre précédemment trouvé. Enfin, nous affichons le plus grand nombre à l'utilisateur.

EXERCICE 4 :

Écrire un programme qui demande à l'utilisateur de saisir 10 nombres réels et les stocke dans un tableau. Ensuite, trouvez et affichez la moyenne des nombres du tableau.
SOLUTION :
#include <stdio.h>
int main() {
int i;
float tableau[10];
float somme = 0, moyenne;
printf("Saisissez 10 nombres réels :\n");
// Saisie des nombres et calcul de la somme
for (i = 0; i < 10; i++) {
printf("Nombre %d : ", i+1);
scanf("%f", &tableau[i]);
somme += tableau[i];
}
// Calcul de la moyenne
moyenne = somme / 10;
printf("La moyenne des nombres saisis est : %.2f\n", moyenne);
return 0;
}
EXPLICATION :
Tout d'abord,
dans ce programme, nous utilisons une boucle for pour demander à l'utilisateur de saisir 10 nombres réels et les stocker dans le tableau tableau[]. En même temps, nous calculons la somme de ces nombres. Ensuite, nous divisons la somme par 10 pour obtenir la moyenne et l'affichons à l'utilisateur. La fonction %.2f est utilisée pour formater l'affichage de la moyenne avec deux décimales.

EXERCICE 5 :

Écrire un programme qui demande à l'utilisateur de saisir 10 caractères et les stocke dans un tableau. Ensuite, affichez ces caractères dans l'ordre inverse.
SOLUTION :
#include <stdio.h>
#define TAILLE_TABLEAU 10
int main() {
char tableau[TAILLE_TABLEAU];
printf("Saisissez 10 caractères :\n");
// Saisie des caractères
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("Caractère %d : ", i+1);
scanf(" %c", &tableau[i]);
}
printf("Les caractères dans l'ordre inverse sont : ");
// Affichage des caractères dans l'ordre inverse
for (int i = TAILLE_TABLEAU - 1; i >= 0; i--) {
printf("%c ", tableau[i]);
}
printf("\n");
return 0;
}
EXPLICATION :
Tout d'abord,
Dans ce programme, nous utilisons une boucle for pour demander à l'utilisateur de saisir 10 caractères et les stocker dans le tableau tableau[]. Ensuite, nous utilisons une autre boucle for pour parcourir le tableau en commençant par la fin, puis affichons les caractères dans l'ordre inverse.

EXERCICE 6:

Écrire un programme en C qui demande à l'utilisateur de saisir un tableau de 5 chaînes de caractères. Ensuite, trouvez et affichez la chaîne de caractères la plus longue dans le tableau.
SOLUTION :
#include <stdio.h>
#include <string.h>
#define TAILLE_TABLEAU 5
#define TAILLE_MAX_CHAINE 100
int main() {
char tableau[TAILLE_TABLEAU][TAILLE_MAX_CHAINE];
char plusLongueChaine[TAILLE_MAX_CHAINE];
int longueurMax = 0;
printf("Saisissez 5 chaînes de caractères :\n");
// Saisie des chaînes de caractères
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("Chaîne %d : ", i+1);
// Lecture caractère par caractère jusqu'au saut de ligne
int index = 0;
char caractere;
while ((caractere = getchar()) != '\n') {
tableau[i][index++] = caractere;
}
tableau[i][index] = '\0'; // Ajoute le caractère de fin de chaîne
}
// Recherche de la chaîne la plus longue
for (int i = 0; i < TAILLE_TABLEAU; i++) {
int longueur = strlen(tableau[i]);
if (longueur > longueurMax) {
longueurMax = longueur;
strcpy(plusLongueChaine, tableau[i]);
}
}
printf("La chaîne la plus longue est : %s\n", plusLongueChaine);
return 0;
}
EXPLICATION :
Tout d'abord,
Dans cette version, nous utilisons la fonction getchar pour lire chaque caractère de la saisie jusqu'à ce que nous atteignions un saut de ligne ('\n'), indiquant la fin de la saisie de la chaîne. Nous stockons les caractères dans le tableau correspondant.
Ensuite,
nous recherchons et affichons la chaîne la plus longue de la même manière que dans la version précédente.
Enfin,
N'hésitez pas à exécuter le programme et à saisir différentes chaînes de caractères pour voir le résultat.

EXERCICE 7:

Écrire un programme qui demande à l'utilisateur de saisir un tableau de 10 nombres entiers. Ensuite, triez les nombres du tableau dans l'ordre croissant et affichez-les.
SOLUTION :
#include <stdio.h>
#define TAILLE_TABLEAU 10
int main() {
int tableau[TAILLE_TABLEAU];
printf("Saisissez %d nombres entiers :\n", TAILLE_TABLEAU);
// Saisie des nombres entiers
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("Nombre %d : ", i+1);
scanf("%d", &tableau[i]);
}
// Tri des nombres dans l'ordre croissant
for (int i = 0; i < TAILLE_TABLEAU - 1; i++) {
for (int j = i + 1; j < TAILLE_TABLEAU; j++) {
if (tableau[i] > tableau[j]) {
int temp = tableau[i];
tableau[i] = tableau[j];
tableau[j] = temp;
}
}
}
printf("Les nombres triés dans l'ordre croissant :\n");
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("%d ", tableau[i]);
}
return 0;
}
EXPLICATION :
Tout d'abord,
Dans ce programme, nous utilisons la fonction scanf pour demander à l'utilisateur de saisir chaque nombre entier du tableau. Ensuite, nous utilisons une méthode de tri à bulles pour trier les nombres dans l'ordre croissant. Enfin, nous affichons les nombres triés à l'écran.

EXERCICE 8:

Écrire un programme qui demande à l'utilisateur de saisir deux tableaux de même taille. Ensuite, additionnez les éléments correspondants des deux tableaux et stockez les résultats dans un troisième tableau. Affichez ensuite le contenu du troisième tableau.
SOLUTION :
#include <stdio.h>
#define TAILLE_TABLEAU 5
int main() {
int tableau1[TAILLE_TABLEAU];
int tableau2[TAILLE_TABLEAU];
int tableauSomme[TAILLE_TABLEAU];
printf("Saisissez %d entiers pour le premier tableau :\n", TAILLE_TABLEAU);
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("Élément %d : ", i+1);
scanf("%d", &tableau1[i]);
}
printf("Saisissez %d entiers pour le deuxième tableau :\n", TAILLE_TABLEAU);
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("Élément %d : ", i+1);
scanf("%d", &tableau2[i]);
}
// Addition des éléments correspondants et stockage dans le troisième tableau
for (int i = 0; i < TAILLE_TABLEAU; i++) {
tableauSomme[i] = tableau1[i] + tableau2[i];
}
printf("Résultat de l'addition des tableaux :\n");
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("%d ", tableauSomme[i]);
}
printf("\n");
return 0;
}
EXPLICATION :
Tout d'abord,
dans ce programme, nous utilisons la fonction scanf pour demander à l'utilisateur de saisir les éléments des deux tableaux. Ensuite, nous additionnons les éléments correspondants des deux tableaux et stockons les résultats dans un troisième tableau tableauSomme. Enfin, nous affichons le contenu du troisième tableau à l'écran.

EXERCICE 9:

Écrire un programme qui demande à l'utilisateur de saisir un tableau de 10 nombres entiers. Ensuite, trouvez et affichez le nombre d'occurrences de chaque élément du tableau.
SOLUTION :
#include <stdio.h>
#define TAILLE_TABLEAU 10
int main() {
int tableau[TAILLE_TABLEAU];
printf("Saisissez %d entiers pour le tableau :\n", TAILLE_TABLEAU);
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("Élément %d : ", i + 1);
scanf("%d", &tableau[i]);
}
int occurrences[TAILLE_TABLEAU] = {0};
// Trouver le nombre d'occurrences de chaque élément
for (int i = 0; i < TAILLE_TABLEAU; i++) {
int element = tableau[i];
occurrences[element]++;
}
// Afficher le nombre d'occurrences de chaque élément
printf("Occurrences des éléments :\n");
for (int i = 0; i < TAILLE_TABLEAU; i++) {
if (occurrences[i] > 0) {
printf("%d : %d\n", i, occurrences[i]);
}
}
return 0;
}
EXPLICATION :
Tout d'abord,
dans ce programme, nous utilisons la fonction scanf pour demander à l'utilisateur de saisir les éléments du tableau. Ensuite, nous créons un tableau occurrences pour stocker le nombre d'occurrences de chaque élément. En parcourant le tableau initial, nous augmentons le compteur correspondant à chaque élément dans le tableau occurrences. Enfin, nous affichons le nombre d'occurrences de chaque élément à l'écran.

EXERCICE 10:

Ecrivez un programme python pour saisir le côté d'un triangle équilatéral et trouvez la surface du triangle donné.
SOLUTION :
#include <stdio.h>
#define TAILLE_TABLEAU 10
int main() {
char tableau[TAILLE_TABLEAU];
int nombreVoyelles = 0;
printf("Saisissez %d caractères pour le tableau :\n", TAILLE_TABLEAU);
for (int i = 0; i < TAILLE_TABLEAU; i++) {
printf("Caractère %d : ", i + 1);
scanf(" %c", &tableau[i]);
}
for (int i = 0; i < TAILLE_TABLEAU; i++) {
char caractere = tableau[i];
if (caractere == 'a' || caractere == 'e' || caractere == 'i' || caractere == 'o' || caractere == 'u' || caractere == 'A' || caractere == 'E' || caractere == 'I' || caractere == 'O' || caractere == 'U') {
nombreVoyelles++;
}
}
printf("Nombre de voyelles dans le tableau : %d\n", nombreVoyelles);
return 0;
}
EXPLICATION :
Tout d'abord,
Dans ce programme, nous utilisons la fonction scanf pour demander à l'utilisateur de saisir les caractères du tableau. En parcourant le tableau, nous vérifions si chaque caractère est une voyelle en comparant avec les voyelles minuscules et majuscules. Si un caractère correspond à une voyelle, nous augmentons le compteur nombreVoyelles. Enfin, nous affichons le nombre de voyelles dans le tableau à l'écran.
